from sys import stdin

def check(k):
    # 两个范围数组，取左边的最大值和右边的最小值判断是否符合条件
    left=[]
    right=[]
    for i in range(n):
        left.append(a[i]-k*b[i])
        right.append(a[i]+k*b[i])
    if max(left)<=min(right):
        return True
    else:
